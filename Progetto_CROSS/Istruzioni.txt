# 1. Compila
javac -d out -cp "lib/gson-2.11.0.jar" $(find src -name "*.java")

# 2. Copia configurazioni
cp src/server/server.properties out/
cp src/client/client.properties out/

# 3. Crea UN SOLO JAR con tutto
cd out
jar cf ../CROSS.jar .
cd ..

# 4. Copia per avere due JAR come richiesto
cp CROSS.jar CROSSServer.jar
cp CROSS.jar CROSSClient.jar

# 5. Esegui
java -cp "CROSSServer.jar:lib/gson-2.11.0.jar" server.ServerMain
java -cp "CROSSClient.jar:lib/gson-2.11.0.jar" client.ClientMain
